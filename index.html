function navigate() { const url = document.getElementById('urlInput').value; if (url) { // URL güvenlik kontrolü if (isSecureUrl(url)) { // URL'yi düzgün formata getir let fullUrl = url; if (!url.startsWith('http://') && !url.startsWith('https://')) { fullUrl = 'https://' + url; } window.location.href = fullUrl; } else { alert('Uyarı: Bu site güvenli görünmüyor!'); } } }
